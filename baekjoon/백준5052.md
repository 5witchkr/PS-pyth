```python
from sys import stdin



count = int(stdin.readline())

for i in range(count):
    n = int(stdin.readline())
    numbers = []
    for ni in range(n):
        numbers.append(stdin.readline().strip())
    numbers.sort()
    isEqual = False
    for j in range(n-1):
        targetSize = len(numbers[j])
        if numbers[j] == numbers[j+1][:targetSize]:
            isEqual = True
            break
    if isEqual:
        print("NO")
    else:
        print("YES")
```
